    contentType: "application/x-www-form-urlencoded; charset=utf-8",//设置编码

        $.each(data , funtion(i, item){
            alert(item[i]);

            alert(item.tagName);
        });
    },
    error : function() {
        alert("删除失败！！");
    }
});
